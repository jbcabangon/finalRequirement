1. Bubble Sort

			FOR i = 0 to n-1
			 FOR j = 0 to n-i-1
			  IF arr[j] > arr[j+1]
			   SWAP

2. Selection Sort

			FOR i = 0 to n-1
			 min = i
			  FOR j = i+1 to n
			   IF arr[j] < arr[min]
			    min = j
			     SWAP

3. Insertion Sort

			FOR i = 1 to n
			 key = arr[i]
			  j = i-1
			   WHILE j >= 0 AND arr[j] > key
			    arr[j+1] = arr[j]
			     INSERT key

4. Linear Search

			FOR each element
			 IF element == key
			  FOUND

5. Binary Search

			SET low, high
			 WHILE low <= high
			  mid = (low + high)/2
			   IF key == arr[mid]
			    FOUND

6. Factorial

			IF n == 0
			 RETURN 1
			  ELSE
			   RETURN n * factorial(n-1)

7. Fibonacci Sequence

			SET a = 0, b = 1
			 FOR i = 1 to n
			  PRINT a
			   a = b
			    b = a + b

8. Palindrome Check

			REVERSE string
			 IF original == reverse
			  PALINDROME

9. Prime Number Check

			FOR i = 2 to n-1
			 IF n % i == 0
			  NOT PRIME

10. Greatest Common Divisor (GCD)

			WHILE b !=0
			 temp = b
			  b = a % b
			   a = temp
